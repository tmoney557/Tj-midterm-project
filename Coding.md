function fizzbuzz() {
  var d = "";
  var displayHTML = "";

  for (i = 1; i < 101; i++) {
    if (i % 3 === 0 && i % 5 === 0) {
      d = "FizzBuzz";
    } else if (i % 5 === 0) {
      d = "Buzz";
    } else if (i % 3 === 0) {
      d = "Fizz";
    } else {
      d = "";
    }

    displayHTML += "<p>" + i + ":" + d + "</p>";
  }
  display.insertAdjacentHTML('afterend', displayHTML); 
}



</script>

</head>

<body onload="fizzbuzz()">

</div>
